## This is a program that makes all text within the specified limits into a rainbow (or user specified color)
// To 'Rainbowify' your text make sure to use namespace Rainbowify
// The following line of code 'Rainbowifies' your text:
Rainbowify.Write("sample text");

// To have a line of text with one color enter the text and rgb code for the color you want in the following line:
Rainbowify.SCWrite("sample text", int, int, int);
// The SC stands for Solid Color
